5.2.1.1. - Ejemplo práctico con NodeJs.

En el siguiente ejemplo se utiliza la librería JWT para el uso del servicio anterior. Se crea el servicio login al que se le pasa un usuario y una contraseña usando el método POST.

var usuario = "paco"
var clave = "secreto"
var privadaJWT = "Mq@45oP!2"
router.post("/login",
  check('usuario').exists().withMessage('Necesario el usuario').customSanitizer((value) => {
    if (value != undefined)
      value = value.replace(regEx_sanityze, "")
    return value;
  }),
  check('clave').exists().withMessage('Necesaria la clave').customSanitizer((value) => {
    if (value != undefined)
      value = value.replace(regEx_sanityze, "")
    return value;
  }),
  (req, res) => {
    const errors = validationResult(req);
    if (!errors.isEmpty()) {
      return res.status(400).json({ errors: errors.array() });
    }
    if (req.body.usuario = usuario && req.body.clave == clave) {
      var token = jwt.sign({ usuario: 'paco' }, privadaJWT, { algorithm: 'HS512', expiresIn: 86400 });
      return res.json(token);
    } else {
      return res.json({ errors: [{ msg: "usuario o clave incorrectos", param: "todos", location: "body" }] })
    }

  })
  • Se comprueba en primer lugar si los parámetros que se le facilitan son válidos y se desinfectan.
    En la línea 20 se comprueba si el usuario y la clave son correctos (en un caso real se obtendrá de una base de datos, otro servicio, LDAP...).
  • En la 21, si es correcto se genera en token, al que se le proporciona los datos en JSON a almacenar en el token, la clave y las opciones que en este caso son el algoritmo y el tiempo de vida del token.
  • En la 22 se envía.

Al enviar los parámetros correctamente se devuelve el token, que la aplicación almacenará para enviarlo en cada petición que lo requiera. Usando POSTMAN se obtiene:

Es necesario por último comprobar cuando se utiliza un servicio protegido (o una página) que se envía el token, que es válido y obtener la información almacenada en el mismo (evitar acceso a la base de datos). Express el framework usado en el ejemplo para Node permite añadir métodos y funciones previos a la ejecución de la petición de forma sencilla, se define la función recursoprotegido que se ejecuta previamente a la validación de datos y al propio método para comprobar si se tiene permisos. Se puede ejecutar en todo el controlado o simplemente en los métodos que lo necesitan:

function recursoprotegido(req, res, next) {
  if (!req.headers.authorization) {
    return res.status(401).json({ errors: [{ msg: "Petición sin token" }] });
  }
  const token = req.headers.authorization.split(" ")[1];
  jwt.verify(token, privadaJWT, (err, decoded) => {
    if (err) {
      return res.status(403).json({ errors: [{ msg: "token incorrecto", param: "token", location: "body" }] })
    } else {
      //se almacena en la petición para su uso en el método
      req.decoded = decoded;
      next();
    }
  });
}
